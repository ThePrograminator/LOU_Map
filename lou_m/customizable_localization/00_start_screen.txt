### Main title of start screen
#Ex: "The Empire of Man, 2017"
defined_text = {
    name = StartScreenTitle
	random = no
	
	text = {
		
        localisation_key = string_screen_title_plural
        trigger = {
			tag_with_plural_name_trigger = yes
        }
    }
    text = {
        localisation_key = string_screen_title
        trigger = {
			always = yes
        }
    }
}
