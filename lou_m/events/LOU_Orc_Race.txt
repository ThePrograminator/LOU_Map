# Infighting Brokeout
country_event = {
	id = LOU_orc_race.1
	title = LOU_orc_race.1.t
	desc = LOU_orc_race.1.d
	picture = ANGRY_MOB_eventPicture

	is_triggered_only = yes
	
	trigger = {
		OR = {
			primary_culture = culture_orkish
			primary_culture = culture_uruks
			primary_culture = culture_half_orc
			primary_culture = culture_ice_orkish
			primary_culture = culture_jungle_orkish
			primary_culture = culture_desert_orkish
		}
		
		religion = religion_orkish
		
		AND = {
			Karma = -33
			NOT = {Karma = 33}
		}
	}
	
	#Punish The Survivers
	option = { 
		name = LOU_orc_race.1.a
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.1 - Option : LOU_orc_race.1.a" }
		
		add_karma = 0.10
		add_yearly_manpower = -0.5
		add_country_modifier = {
			name = orc_religion_fuel_of_industry
			duration = 3650
		}
	}
	
	#Reward The Victors
	option = { 
		name = LOU_orc_race.1.b
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.1 - Option : LOU_orc_race.1.b" }

		add_karma = -0.10
		add_prestige = -10
		add_country_modifier = {
			name = orc_religion_reward_victors
			duration = 3650
		}
	}
}

# raiding parties
country_event = {
	id = LOU_orc_race.6
	title = LOU_orc_race.6.t
	desc = LOU_orc_race.6.d
	picture = BORDER_TENSION_eventPicture

	is_triggered_only = yes
	
	trigger = {
	
		AND = {
			OR = {
				primary_culture = culture_orkish
				primary_culture = culture_uruks
				primary_culture = culture_half_orc
				primary_culture = culture_ice_orkish
				primary_culture = culture_jungle_orkish
				primary_culture = culture_desert_orkish
			}
			religion = religion_orkish
			is_at_war = no
			
			NOT = {
				karma = -33
			}
			
			NOT = {
				has_country_modifier = orc_religion_unguide_flames
			}
		}
	
		any_owned_province = {
			has_owner_culture = yes
			has_owner_religion = yes
			any_neighbor_province = {
				
				NOT = { 
					owned_by = ROOT 
					culture = culture_orkish
					culture = culture_uruks
					culture = culture_half_orc
					culture = culture_ice_orkish
					culture = culture_jungle_orkish
					culture = culture_desert_orkish
					religion = religion_orkish
					culture = culture_spider
					religion = religion_spider
					has_province_modifier = orc_religion_raided_neighbour_province
				}
				owner = {
					NOT = { war_with = ROOT }
				}
			}
		}
		
	}
	
	#Let Them Raid
	option = { 
		name = LOU_orc_race.6.a
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.6 - Option : LOU_orc_race.6.a" }
		
		add_karma = -0.10
		
		random_owned_province = {
			limit = {
				has_owner_culture = yes
				has_owner_religion = yes
				any_neighbor_province = {
					#NOT = { owned_by = ROOT }
					NOT = { 
						owned_by = ROOT 
						culture = culture_orkish
						culture = culture_uruks
						culture = culture_half_orc
						culture = culture_ice_orkish
						culture = culture_jungle_orkish
						culture = culture_desert_orkish
						religion = religion_orkish
						culture = culture_spider
						religion = religion_spider
						has_province_modifier = orc_religion_raided_neighbour_province
					}
					owner = {
						NOT = { war_with = ROOT }
					}
				}
			}
			random_neighbor_province = {
				limit = {
					NOT = {
						owned_by = ROOT 
						
						culture = culture_orkish
						culture = culture_uruks
						culture = culture_half_orc
						culture = culture_ice_orkish
						culture = culture_jungle_orkish
						culture = culture_desert_orkish
						religion = religion_orkish
						
						culture = culture_spider
						religion = religion_spider
						has_province_modifier = orc_religion_raided_neighbour_province
					}
					owner = {
						NOT = { war_with = ROOT }
					}
				}
				
				random_list = {
					33 = {
						add_base_tax = -1
						FROM = {
							add_adm_power = 25
						}
					}
					33 = {
						add_base_production = -1
						FROM = {
							add_dip_power = 25
						}
					}
					33 = {
						add_base_manpower = -1
						FROM = {
							add_mil_power = 25
						}
					}
				}
				
				add_devastation = -100
				
				remove_loot = {
					who = ROOT
					amount = 100
				}
				
				add_province_modifier = {
					name = orc_religion_raided_neighbour_province 
					duration = 3650
				}
				
				owner = {
					add_opinion = {
						who = ROOT
						modifier = orc_raided_province
						years = 25
					}
				}
				
				hidden_effect = {
					THIS = {
						province_event = {
							id = LOU_orc_race.7
						}
					}
				}
			}
		}
	}
	
	#We haven't ordered a raid!
	option = { 
		name = LOU_orc_race.6.b
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.6 - Option : LOU_orc_race.6.b" }
		
		add_karma = 0.10
		
		add_prestige = -10
		add_manpower = -2
		
		add_country_modifier = {
			name = orc_religion_unguide_flames
			duration = 3650
		}
	}
}

country_event = {
	id = LOU_orc_race.8
	title = LOU_orc_race.8.t
	desc = LOU_orc_race.8.d
	picture = ACCUSATION_eventPicture

	is_triggered_only = yes
	
	trigger = {
		AND = {
			OR = {
				primary_culture = culture_orkish
				primary_culture = culture_uruks
				primary_culture = culture_half_orc
				primary_culture = culture_ice_orkish
				primary_culture = culture_jungle_orkish
				primary_culture = culture_desert_orkish
			}
			religion = religion_orkish
			
			NOT = {
				karma = -33
			}
			
			
			NOT = {
				has_country_modifier = orc_religion_fuel_of_war
				has_country_modifier = orc_religion_spark_of_industry
			}
		}
		
		any_known_country = {
			NOT = {
				primary_culture = culture_spider
				religion = religion_spider
				has_country_modifier = orc_religion_frighten_traders
			}
			any_active_trade_node = {
				has_merchant = PREV
				any_trade_node_member_country = { tag = ROOT }
			}
		}
	}
	
	#This will hurt us
	option = { 
		name = LOU_orc_race.8.a
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.8 - Option : LOU_orc_race.8.a" }
		
		random_known_country = {
			limit = {
				NOT = {
					primary_culture = culture_spider
					religion = religion_spider
					has_country_modifier = orc_religion_frighten_traders
				}
			
				any_active_trade_node = {
					has_merchant = PREV
					any_trade_node_member_country = { tag = ROOT }
				}
			}
			add_mercantilism = -2
			kill_advisor = random
			
			add_country_modifier = {
				name = orc_religion_frighten_traders
				duration = 3650
			}
			
			add_opinion = {
				who = ROOT
				modifier = orc_killed_merchants
				years = 10
			}
			
			hidden_effect = {
				country_event = {
					id = LOU_orc_race.9
				}
			}
		}
		
		ROOT = {
			add_karma = -0.10
		
			add_country_modifier = {
				name = orc_religion_fuel_of_war 
				duration = 3650
			}
		}
	}
	
	#Compensate
	option = { 
		name = LOU_orc_race.8.b
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - LOU_orc_race.8 - Option : LOU_orc_race.8.b" }
		
		add_adm_power = -50
		add_dip_power = -50
		
		add_prestige = -10
		
		add_treasury = -100
		
		ROOT = {
			add_karma = 0.10
		
			add_country_modifier = {
				name = orc_religion_spark_of_industry 
				duration = 3650
			}
		}
	}
}
